factory NoiseTable.fromSeed([int? seed]) {
  final seed_ = seed ?? Xoshiro128.randomSeed();
  final real = Uint8List(16)..[0] = 1;
  for (var i = 1; i < 4; i++) {
    real[i * 4] = seed_;
    real[i * 4 + 1] = seed_ >> 8;
    real[i * 4 + 2] = seed_ >> 16;
    real[i * 4 + 3] = seed_ >> 24;
  }
  final rng = Xoshiro128P.fromSeed(real);
  return NoiseTable(rng);
}
