Returns the 2-dimensuonal gradient vector at index, between 0 and 7.

Any index out of bounds is wrapped around using index % 8.

Implementation

Vec2 gradient2d(int index) {
  const diag = sqrt1_2;
  return switch (index % 8) {
    0 => const (1.0, 0.0),
    1 => const (-1.0, 0.0),
    2 => const (0.0, 1.0),
    3 => const (0.0, -1.0),
    4 => const (diag, diag),
    5 => const (-diag, diag),
    6 => const (diag, -diag),
    7 => const (-diag, -diag),
    _ => throw StateError('Unreachable'),
  } as Vec2;
}
